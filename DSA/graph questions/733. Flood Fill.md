class Solution {
   public:
    int m, n;
    bool isSafe(int i, int j) {
        return i >= 0 && i < m && j >= 0 && j < n;
    }
    vector<vector<int>> floodFill(vector<vector<int>>& image, int sr, int sc, int color) {
        m = image.size();
        n = image[0].size();
        queue<pair<int, int>> que;
        vector<vector<int>> directions = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        que.push({sr, sc});

        
        // current image color is old color -> save it
        int oldColor = image[sr][sc]; 
        // new color is proposed color which we want to fill
        int newColor = color; 
        
        // #note: old color can be any color (1 is just an example)

        if (oldColor == newColor) {
            return image;
        }

        while (!que.empty()) {
            int i = que.front().first;
            int j = que.front().second;
            que.pop();

            // basic bfs order traversal
            for (auto& dir : directions) {
                int _i = dir[0] + i;
                int _j = dir[1] + j;

                if (isSafe(_i, _j) && image[_i][_j] == oldColor) {
                    que.push({_i, _j});
                    image[_i][_j] = newColor;
                }
            }
        }

        // fill source(center) with new color
        if (image[sr][sc] != color) {
            image[sr][sc] = color;
        }

        return image;
    }
};
